#MULT: MACRO D, A1, P
	LDA A1 ;load the first parameter into the accumulator
	LDX #1 ;will be used to count to P

L1 	CMP P,X ; compare P with X
	BLS MUL ;branching to MUL as many times as the value of P
	STA D ; store the result in D
	BRA ENDPRG ;ending the loop

MUL ASL ;multiply by 2
	INX ;increment X
	BRA L1 

ENDPRG END

#ENDM
